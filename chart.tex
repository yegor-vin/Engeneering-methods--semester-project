\section*{Key Insights}
\begin{itemize}
\item \textbf{No central servers} = Zero scaling costs
\item \textbf{Community editing} = Free timetable maintenance
\item \textbf{Open formats} = No licensing fees
\item \textbf{Offline routing} = No real-time data expenses
\end{itemize}
